## What is Typescript

Typescript is a superset of javascript

```ts
import { useEffect, useState } from "react";

const Home = ({ data }: { data: any }) => {
  const [user, setUser] = useState(null);

  useEffect(() => {
    const res = await fetch("https://ssr.com");
    setUser(res.json());
    return user;
  }, [data]);

  return (
    <div>
      <h1>Hello {user}</h1>
    </div>
  );
};
```
